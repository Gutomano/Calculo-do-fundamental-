{

        gt = ler.nextInt();

     media = ((gtr+gt+ferrari)/3);
     
     if (media >= 8 )

    }

    {
        System.out.println(" Voce fara recuperacao ");

        System.out.println("puts voce infelizmete foi reprovado");

    }
}
